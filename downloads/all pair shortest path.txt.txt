#include <stdio.h>
#include <limits.h>

#define n 4

void print(int arr[][n]) {
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            printf("%d , ", arr[i][j]);
        }
        printf("\n");
    }
}

int min(int x, int y) {
    return (x < y) ? x : y;
}

void allpairshortestpath(int graph[][n]) {
    for(int k = 0; k < n; k++) {
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                if(graph[i][k] != INT_MAX && graph[k][j] != INT_MAX) {
                    graph[i][j] = min(graph[i][j], graph[i][k] + graph[k][j]);
                }
            }
        }
    }
    printf("Calculated matrix\n");
    print(graph);
}

int main() {
    int graph[n][n] = {{0, 3, INT_MAX, 7},
                       {8, 0, 2, INT_MAX},
                       {5, INT_MAX, 0, 1},
                       {2, INT_MAX, INT_MAX, 0}};
    
    printf("Original matrix\n");
    print(graph);
    allpairshortestpath(graph);
    return 0;
}
